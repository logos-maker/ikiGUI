54 mywin->frame.w = w;
55 mywin->frame.h = h;
56 mywin->dis=XOpenDisplay((char *)0); // Get the display

59 int y = 0;
60 unsigned long black = BlackPixel(mywin->dis,mywin->screen);// get the color black

63
64 // To get event from closing window
65 mywin->wm_delete_window = XInternAtom(mywin->dis, "WM_DELETE_WINDOW", False);
66 XSetWMProtocols(mywin->dis, mywin->win, &mywin->wm_delete_window, 1);
67
68 // Event types to recive
69 XSelectInput(mywin->dis, mywin->win, ExposureMask|PointerMotionMask|ButtonPressMask|ButtonReleaseMask|KeyPressMask|KeyReleaseMask|FocusChangeMask|EnterWindowMask|LeaveWindowMask); // Select event types to get
70
71 mywin->gc=DefaultGC(mywin->dis,mywin->screen);
72
73 XMapRaised(mywin->dis, mywin->win);
74
75 XWindowAttributes wa = {0};
76 XGetWindowAttributes(mywin->dis, mywin->win, &wa);
77
78 mywin->frame.pixels = (unsigned int *)malloc(w * h * 4);
79 mywin->bitmap = XCreatePixmap(mywin->dis, mywin->win, w, h, 1);
80 mywin->image = XCreateImage(mywin->dis, wa.visual, wa.depth, ZPixmap, 0, (char*)mywin->frame.pixels, w, h, 32, w * 4);

82 XReparentWindow(mywin->dis, mywin->win,(Window)ptr, 0, 0);

95void ikigui_get_events(ikigui_window *mywin){
96 // values for recognicing changes in mousemovements and mouse buttons.
97
98 mywin->mouse.old_button_press = mywin->mouse.buttons; // old value for buttons. For finding changes later on.
99 mywin->mouse.old_x = mywin->mouse.x ; // old value for x coordinate.
100 mywin->mouse.old_y = mywin->mouse.y ; // old value for y coodrinate.
101
102 while( XPending(mywin->dis) > 0 ){ // no of events in que
103 XNextEvent(mywin->dis, &mywin->event); // Get next event
104
105 if (mywin->event.type== ClientMessage){ // User Closes window
106 if ((Atom) mywin->event.xclient.data.l[0] == mywin->wm_delete_window) {
107 ikigui_close_window(mywin);
108 exit(1);
109 }
110 }
111 if (mywin->event.type==Expose && mywin->event.xexpose.count==0) ikigui_update_window(mywin); // Window graphics needs a redraw
112 if (mywin->event.type==KeyPress&& XLookupString(&mywin->event.xkey,mywin->text,255,&mywin->key,0)==1) { // the XLookupString routine to converts the KeyPress event data into regular text.
113 //printf("You pressed the %c key!\n",mywin.text[0]);
114 }
115
116 switch(mywin->event.type){
117 case MotionNotify:{// Mouse movement
118 mywin->mouse.x_rel = mywin->event.xmotion.x -old_x ;
119 mywin->mouse.y_rel = mywin->event.xmotion.y -old_y ;
120
121 old_x = mywin->mouse.x = mywin->event.xmotion.x ;
122 old_y = mywin->mouse.y = mywin->event.xmotion.y ;
123 }
124 break;
125 case ButtonPress: //printf("Mouse botton down x %d y %d ",mywin.event.xbutton.x,mywin.event.xbutton.y);
126 switch(mywin->event.xbutton.button){
127 case 1: mywin->mouse.buttons |= MOUSE_LEFT; break; // printf("Left\n"); break;
128 case 2: mywin->mouse.buttons |= MOUSE_MIDDLE; break; // printf("Middle\n");break;
129 case 3: mywin->mouse.buttons |= MOUSE_RIGHT; break; // printf("Right\n");break;
130 //default: printf("???%d",mywin.event.xbutton.button);break;
131 }
132 break;
133 case ButtonRelease: //printf("Mouse botton up x %d y %d \n",mywin.event.xbutton.x,mywin.event.xbutton.y); break;
134 switch(mywin->event.xbutton.button){
135 case 1: mywin->mouse.buttons &= ~MOUSE_LEFT; break; // printf("Left\n"); break;
136 case 2: mywin->mouse.buttons &= ~MOUSE_MIDDLE; break; // printf("Middle\n");break;
137 case 3: mywin->mouse.buttons &= ~MOUSE_RIGHT; break; // printf("Right\n");break;
138 //default: printf("???%d",mywin.event.xbutton.button);break;
139 }
140 break;
141 }
142 } // while loop end
143
144 mywin->mouse.left_click = (mywin->mouse.old_button_press == 0) && (mywin->mouse.buttons & MOUSE_LEFT);
145 mywin->mouse.middle_click = (mywin->mouse.old_button_press == 0) && (mywin->mouse.buttons & MOUSE_MIDDLE);
146 mywin->mouse.right_click = (mywin->mouse.old_button_press == 0) && (mywin->mouse.buttons & MOUSE_RIGHT);
147 mywin->mouse.left_release = (mywin->mouse.old_button_press == 1) && (!(mywin->mouse.buttons & MOUSE_LEFT));
148 mywin->mouse.middle_release = (mywin->mouse.old_button_press == 1) && (!(mywin->mouse.buttons & MOUSE_MIDDLE));
149 mywin->mouse.right_release = (mywin->mouse.old_button_press == 1) && (!(mywin->mouse.buttons & MOUSE_RIGHT));
150};
