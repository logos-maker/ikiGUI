48LRESULT CALLBACK WindowProcessMessage(HWND window_handle, UINT message, WPARAM wParam, LPARAM lParam) {
49
50 ikigui_window *mywin = (ikigui_window*)GetWindowLongPtr(window_handle, GWLP_USERDATA);
51
52 switch(message) {
53 case WM_CLOSE: break;// Someone has pressed x on window
54 case WM_QUIT: break;// This function is usually called when the main window receives WM_DESTROY
55 case WM_DESTROY: DestroyWindow(window_handle); break;// Window is about to be destroyed, and can not be hindered.
56 case WM_PAINT: {
57 PAINTSTRUCT paint;
58 HDC device_context;
59 device_context = BeginPaint(window_handle, &paint);
60 BitBlt(device_context, paint.rcPaint.left, paint.rcPaint.top, paint.rcPaint.right - paint.rcPaint.left, paint.rcPaint.bottom - paint.rcPaint.top, mywin->bitmap_device_context, paint.rcPaint.left, paint.rcPaint.top, SRCCOPY);
61 EndPaint(window_handle,&paint);
62 } break;
63 case WM_KILLFOCUS: mywin->has_focus = false; break;
64 case WM_SETFOCUS: mywin->has_focus = true; break;
65
66 case WM_KEYDOWN: break;
67 case WM_KEYUP: break;
68
69 case WM_MOUSEMOVE: { POINT pos;
70 if(mywin->mouse.buttons == MOUSE_LEFT){ // SetCursorPos();
71 GetCursorPos(&pos); // absolute coordinate on screen
72 mywin->mouse.y_intern += pos.y - mywin->mouse.pos.y;
73 mywin->mouse.pos.y = pos.y;
74 mywin->mouse.x_intern += pos.x - mywin->mouse.pos.x;
75 mywin->mouse.pos.x = pos.x;
76
77 }else{
78 mywin->mouse.x_intern = LOWORD(lParam); // coordniate relative to window
79 mywin->mouse.y_intern = HIWORD(lParam);
80 }
81 } break;
82 case WM_MOUSELEAVE: break;
83 case WM_LBUTTONDOWN: mywin->mouse.buttons_intern |= MOUSE_LEFT; SetCapture(window_handle); GetCursorPos(&mywin->mouse.pos);break;
84 case WM_LBUTTONUP: mywin->mouse.buttons_intern &= ~MOUSE_LEFT; ReleaseCapture();break;

133void ikigui_get_events(struct ikigui_window *mywin){
134
135
136 while(PeekMessage(&mywin->message, NULL, 0, 0, PM_REMOVE)) { DispatchMessage(&mywin->message); }
137
138 // values for recognicing changes in mousemovements and mouse buttons.
139 mywin->mouse.old_x = mywin->mouse.x ; // old value for x coordinate.
140 mywin->mouse.old_y = mywin->mouse.y ; // old value for y coodrinate.
141 mywin->mouse.x=mywin->mouse.x_intern;
142 mywin->mouse.y=mywin->mouse.y_intern;
143 mywin->mouse.old_button_press = mywin->mouse.buttons; // old value for buttons. For finding changes later on.
144 mywin->mouse.buttons = mywin->mouse.buttons_intern;
145 mywin->mouse.left_click = (mywin->mouse.old_button_press == 0) && (mywin->mouse.buttons & MOUSE_LEFT);
146 mywin->mouse.middle_click = (mywin->mouse.old_button_press == 0) && (mywin->mouse.buttons & MOUSE_MIDDLE);
147 mywin->mouse.right_click = (mywin->mouse.old_button_press == 0) && (mywin->mouse.buttons & MOUSE_RIGHT);
148 mywin->mouse.left_release = (mywin->mouse.old_button_press == 1) && (!(mywin->mouse.buttons & MOUSE_LEFT));
149 mywin->mouse.middle_release = (mywin->mouse.old_button_press == 1) && (!(mywin->mouse.buttons & MOUSE_MIDDLE));
150 mywin->mouse.right_release = (mywin->mouse.old_button_press == 1) && (!(mywin->mouse.buttons & MOUSE_RIGHT));
151
152}
