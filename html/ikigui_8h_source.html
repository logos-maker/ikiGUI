52 display->map = (char*)calloc(columns*rows,sizeof(char));

60 display->columns = columns ; // Set to a default value

89int ikigui_mouse_pos(struct ikigui_map *display, int x, int y){ // returns -1 if outside the character display
90 if( (x < 0) || (y < 0) ) return -1; // To the left or over the map
91 if( ( x < (( display->x_spacing) * display->columns))// To the right of the map
92 && ( y < (( display->y_spacing) * display->rows)) // below the map
93 ){
94 int col = x / display->x_spacing ;
95 int row = y / display->y_spacing ;
96 if( (x < ( col * display->x_spacing +display->char_width)) && (y < ( row * display->y_spacing +display->char_hight)) ) // inside the tile
97 return (col + (row * display->columns)); // Returns the index of the character in your character array.
98 };
99 return -1; // Pixel coordinate is outside of the character display.

106void ikigui_map_draw(struct ikigui_map *display, char filling, int x, int y){ // x y is pixel coordinate to draw it in the window
107
108 ikigui_rect srcrect = { .w = display->char_width, .h = display->char_hight }; // , .x = 0, .y = 0, } ;
109 ikigui_rect dstrect = { .w = display->char_width, .h = display->char_hight };
110
111 int set_w;
112 if(display->direction) set_w = srcrect.w ;
113 else set_w = srcrect.h ;
114
115 for(int i = 0 ; i < display->rows ; i++ ){ // draw all rows
116 dstrect.y = i * display->y_spacing + y;
117 for(int j = 0 ; j < display->columns ; j++ ){ // draw all columns
118 dstrect.x = (j * display->x_spacing) + x;
119 int val = set_w * (display->map[i*display->columns + j] + display->offset) ;
120 if(val<0) continue ; // Don't draw tile if given lower value than 0. // val=0;
121 if(display->direction)srcrect.x = val ; else srcrect.y = val ;
122 switch(filling){ // Draw the character buffer to window.
123 case 0: ikigui_blit_alpha (display->renderer,display->texture, dstrect.x, dstrect.y, &srcrect); break;
124 case 1: ikigui_blit_filled(display->renderer,display->texture, dstrect.x, dstrect.y, &srcrect); break;
125 case 2: ikigui_blit_fast (display->renderer,display->texture, dstrect.x, dstrect.y, &srcrect); break;
126 case 3: ikigui_blit_hollow(display->renderer,display->texture, dstrect.x, dstrect.y, &srcrect); break;
127 }
128 }
129 }
