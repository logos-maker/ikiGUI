7unsigned int alpha_channel(unsigned int color,unsigned int temp){ // Internal for usage in other functions // done with fixed point math
8 unsigned char alpha = temp >> 24; // Alpha channel
9 unsigned char alpha_inv = ~alpha;
10 unsigned char rf = (temp&0xff0000)>>16; // Red forground
11 unsigned char gf = (temp&0xff00)>>8; // Green forground
12 unsigned char bf = temp&0xff; // Blue forground
13 unsigned char rb = (color&0xff0000)>>16; // Red beckground
14 unsigned char gb = (color&0xff00)>>8; // Red background
15 unsigned char bb = color&0xff; // Blue background
16 unsigned char ro = (alpha_inv*rb + alpha*rf)>>8; // background + forground
17 unsigned char go = (alpha_inv*gb + alpha*gf)>>8; // background + forground
18 unsigned char bo = (alpha_inv*bb + alpha*bf)>>8; // background + forground
19 return (unsigned int)((ro << 16) + (go<< 8) + bo);
20}
